// Diccionario de rutas que un administrador o desarrollador suele dejar expuestas
const rutasOcultas = [
    "/admin", "/v1/api", "/config", "/backup.sql", "/.env", 
    "/api/users", "/dashboard", "/phpmyadmin", "/wp-admin", "/test"
];

async function buscarRutasOcultas() {
    console.log("рҹ”Қ Escaneando directorios del proyecto...");

    for (let ruta of rutasOcultas) {
        // Construimos la URL completa basada en tu dominio actual
        const urlObjetivo = window.location.origin + ruta;

        try {
            const respuesta = await fetch(urlObjetivo, { method: 'GET' });

            // Si devuelve 200 (Existe) o 403 (Existe pero no hay permiso), la ruta estГЎ ahГӯ.
            if (respuesta.status === 200) {
                console.log(вҒ вҖҜ%c[ВЎENCONTRADO (200)!] Ruta expuesta: ${urlObjetivo}вҖҜвҒ , 'background: #222; color: #00ff00; font-size: 14px;');
            } else if (respuesta.status === 403) {
                console.log(вҒ вҖҜ%c[AVISO (403)] La ruta existe pero estГЎ restringida: ${urlObjetivo}вҖҜвҒ , 'color: #ffcc00;');
            }
        } catch (error) {
            // Ignoramos errores de conexiГіn
        }
    }
    console.log("рҹҸҒ Escaneo de rutas finalizado.");
}

buscarRutasOcultas();

const urlProtegida = window.location.origin + "/api/perfil-privado";

async function forzarTokensSesion() {
    console.log("рҹ•өпёҸвҖҚвҷӮпёҸ Iniciando fuerza bruta de tokens de sesiГіn...");

    // Simulamos probar tokens correlativos del 1000 al 1050
    for (let tokenAnГіmalo = 1000; tokenAnГіmalo <= 1050; tokenAnГіmalo++) {
        try {
            // Enviamos la peticiГіn incluyendo el token en los encabezados
            // (Simulando que cambiamos nuestra cookie de sesiГіn)
            const respuesta = await fetch(urlProtegida, {
                method: 'GET',
                headers: {
                    'Authorization': вҒ вҖҜBearer TOKEN_SIMULADO_${tokenAnГіmalo}вҖҜвҒ 
                }
            });

            // Si el servidor responde con 200, significa que adivinamos un token activo
            if (respuesta.status === 200) {
                console.log(вҒ вҖҜ%c[ВЎГүXITO!] Token vГЎlido encontrado: TOKEN_SIMULADO_${tokenAnГіmalo}вҖҜвҒ , 'background: #222; color: #00ff00; font-size: 14px;');
                break; 
            } else {
                console.log(вҒ вҖҜToken ${tokenAnГіmalo}: Acceso Denegado (${respuesta.status})вҖҜвҒ );
            }
        } catch (error) {
            console.error("Error en la prueba:", error);
        }
    }
    console.log("рҹҸҒ Prueba de tokens finalizada.");
}
